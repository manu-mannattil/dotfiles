% Custom preamble for latexdiff.
% To use this, run latexdiff as
%   latexdiff -p .../path/to/preamble.tex <arg>...

\usepackage[normalem]{ulem}
\def\ULthickness{0.75pt}
\usepackage{color}

\definecolor{CustomColorDEL}{rgb}{0.9,0.1,0.1}
\definecolor{CustomColorADD}{rgb}{0.1,0.1,0.9}

\providecommand{\DIFadd}[1]{{\protect\color{CustomColorADD}\uline{#1}}}
\providecommand{\DIFdel}[1]{{\protect\color{CustomColorDEL}\sout{#1}}}

\providecommand{\DIFaddbegin}{}
\providecommand{\DIFaddend}{}
\providecommand{\DIFdelbegin}{}
\providecommand{\DIFdelend}{}
\providecommand{\DIFmodbegin}{}
\providecommand{\DIFmodend}{}

% Floats.
\providecommand{\DIFaddFL}[1]{\DIFadd{#1}}
\providecommand{\DIFdelFL}[1]{\DIFdel{#1}}
\providecommand{\DIFaddbeginFL}{}
\providecommand{\DIFaddendFL}{}
\providecommand{\DIFdelbeginFL}{}
\providecommand{\DIFdelendFL}{}

\makeatletter

% Force hyperref to avoid color links.
\@ifpackageloaded{hyperref}{\hypersetup{hidelinks}}{}

% Put line numbers if we're not already doing so.
\@ifpackageloaded{lineno}{}{
  \usepackage[columnwise]{lineno}
  \if@twocolumn
    \setlength\linenumbersep{2pt}
  \else
    \setlength\linenumbersep{10pt}
  \fi
  % Same thing as above, but for RevTeX.
  \twocolumn@sw{\setlength\linenumbersep{2pt}}{\setlength\linenumbersep{10pt}}
  \definecolor{CustomColorSnow}{rgb}{0.5,0.5,0.5}
  \def\linenumberfont{\color{CustomColorSnow}\normalfont\footnotesize\sffamily}
  \linenumbers
}

\makeatother
